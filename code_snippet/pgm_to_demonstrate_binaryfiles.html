C program to demonstrate binary files.

      #include<stdio.h>
      struct customer
      {
          int accno;
          char name[50];
          float balance;
      };

      void main()
      {
          struct customer cust;
          FILE *fp;
          fp=fopen("bank.dat","wb");
          for(;;)
          {
              printf("\nEnter account no(0 to stop):");
              scanf("%d",&cust.accno);

              if(cust.accno==0) break;
              fflush(stdin);
              printf("\nEnter customer name:");
              gets(cust.name);
              printf("\nEnter balance amount:");
              scanf("%f",&cust.balance);
              fwrite(&cust,sizeof(cust),1,fp);
          }
          printf("\nbank.dat file is created.");
          fclose(fp);
       }
