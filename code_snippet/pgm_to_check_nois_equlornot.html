C Program to determine the number is equal or not. 

      #include<stdio.h>
      #include<conio.h>
      void main()
      {
          int n,r1,r2,r3,p,m;
          printf("Enter The Four Digit Number:");
          scanf("%d", &n);
          m=n;
          r1=n%10;
          n=n/10;
          r2=n%10;
          n=n/10;
          r3=n%10;
          n=n/10;
          p=(r1*1000)+(r2*100)+(r3*10)+n;
          if(p==m)
          printf("The Numbers are equal");
          else
              printf("The Numbers are not equal");
      }
