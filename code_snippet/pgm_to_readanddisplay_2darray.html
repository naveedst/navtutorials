C Program to read and display 2D array with functions.

      #include<stdio.h>
      #define ROWS 50
      #define COLS 50

      void getarr(int arr[ROWS][COLS],int r,int c)
      {
          int i,j;
          printf("\n");
          for(i=0;i<r;i++)
          {
              for(j=0;j<c;j++)
              {
                  scanf("%d",&arr[i][j]);
              }
          }
      }

      void disparr(int arr[ROWS][COLS],int r,int c)
      {
          int i,j;
          for(i=0;i<r;i++)
          {
              for(j=0;i<c;j++)
              {
                  printf("%d\t",arr[i][j]);
              }
              printf("\n");
          }
      }

      void main()
      {
          int x[ROWS][COLS];
          int r,c;

          printf("How many rows, cols?");
          scanf("%d,%d",&r,&c);
          printf("\nEnter matrix:");

          getarr(x,r,c);
          printf("\nMatrix is:");
          disparr(x,r,c);
      }

Output:

    How many elements? 2,3
    Enter matrix:
    1 2 3
    4 5 6

    Matrix is:
    1 2 3
    4 5 6
